//Introduction more "atomic" programming with functions //We will use a function or collection of reusable commands to //draw a geometric pattern resembling a face function Blueskull(x,y){ stroke(255,255,255,255); fill(150,200,255,255); ellipse(x+30,y+20,70,70); rect(x+5, y+5, 20, 20, 10, 10); rect(x+35, y+5, 20, 20, 10, 10); rect(x+10, y+40, 10, 20, 5, 5); rect(x+25, y+40, 10, 20, 5, 5); rect(x+40, y+40, 10, 20, 5, 5); } function setup() { createCanvas(800,500); background(255,255,100,255); //this is a color in RGBA colorspace : red, green, blue and alpha (opacity) for(var counter=0;counter<8;counter++){ Blueskull(15+counter*80, 40); Blueskull(50+counter*80, 120); Blueskull(80+counter*80, 200); Blueskull(110+counter*80, 280); Blueskull(160+counter*80, 360); } }
